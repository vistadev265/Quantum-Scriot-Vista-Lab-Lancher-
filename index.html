import pygame import random # Initialize Pygame pygame.init() # Get the system's display size display_info = pygame.display.Info() WINDOW_WIDTH = display_info.current_w WINDOW_HEIGHT = display_info.current_h # Set up the window in fullscreen mode window = pygame.display.set_mode((WINDOW_WIDTH, WINDOW_HEIGHT), pygame.FULLSCREEN) pygame.display.set_caption('Vista Beat') # Colors WHITE = (255, 255, 255) RED = (255, 0, 0) BLUE = (0, 0, 255) # Fonts font_large = pygame.font.Font(None, 48) font_medium = pygame.font.Font(None, 36) font_small = pygame.font.Font(None, 24) # Game variables score = 0 clock = pygame.time.Clock() cubes = [] cube_speed = 8 cube_spawn_frequency = 999 # Function to generate a new cube def generate_cube(): size = random.randint(20, 50) x = random.randint(size, WINDOW_WIDTH - size) y = -size color = random.choice([RED, BLUE]) cubes.append({'rect': pygame.Rect(x, y, size, size), 'color': color}) # Function to draw text on screen def draw_text(text, font, color, x, y): text_surface = font.render(text, True, color) text_rect = text_surface.get_rect() text_rect.center = (x, y) window.blit(text_surface, text_rect) # Main game loop running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.MOUSEBUTTONDOWN: mouse_x, mouse_y = pygame.mouse.get_pos() if event.button == 1: # Left mouse button for cube in cubes[:]: if cube['color'] == RED and cube['rect'].collidepoint(mouse_x, mouse_y): cubes.remove(cube) score += 1 elif event.button == 3: # Right mouse button for cube in cubes[:]: if cube['color'] == BLUE and cube['rect'].collidepoint(mouse_x, mouse_y): cubes.remove(cube) score += 1 # Clear the screen window.fill((0, 0, 0)) # Update game state # Spawn new cubes generate_cube() # Move cubes for cube in cubes: cube['rect'].move_ip(0, cube_speed) # Check for collisions with cubes for cube in cubes[:]: if cube['rect'].bottom >= WINDOW_HEIGHT: cubes.remove(cube) # Draw objects for cube in cubes: pygame.draw.rect(window, cube['color'], cube['rect']) # Draw score score_text = font_small.render('Score: ' + str(score), True, WHITE) window.blit(score_text, (10, 10)) # Update the display pygame.display.flip() # Cap the frame rate clock.tick(30) # Quit Pygame pygame.quit() sys.exit()
